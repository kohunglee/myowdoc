如何测试反复增减动态模型？

```js
let i = 0;
const maxCount = 300;
var inSetIndex = -1;

const timer = setInterval(() => {
    // 1. 你的逻辑
    if(inSetIndex === -1){
        inSetIndex = dataProc.process(k.tempData, 'texture_0', {x:0}, dls);
    } else {
        k.deleteModBlock(inSetIndex);
        inSetIndex = -1;
    }
    
    // 2. 计数器增加
    i++;

    // 3. 判断是否结束
    if (i >= maxCount) {
        clearInterval(timer); // 清除定时器
        console.log("完成");
    } else {
        console.log('第' + i + '次');
    }
}, 100); // 每隔 1 秒触发一次
```
